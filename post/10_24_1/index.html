#include <iostream>
#include <vector>
#include <iterator>
using namespace std;

typedef int T;

struct Node
{
	T data;
	Node* next;
	Node(const T& d) : data(d), next(NULL) {}
	operator T() { return data; }  // 类型转换函数将Node 转换成int并能实现输出
};

int main()
{
	Node a(10), b(20), c(30), d(40), e(50);
	a.next = &b;
	b.next = &c;
	c.next = &d;
	e.next = b.next;  //将e插入和b之间 必须先执行后面的节点插入 
	b.next = &e;
	Node* p = &a; //P为头指针
	while (p != NULL)
	{
		cout << *p << endl;
		p = (*p).next;
	}

}
